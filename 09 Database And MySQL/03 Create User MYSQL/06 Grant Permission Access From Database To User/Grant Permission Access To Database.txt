GRANT PERMISSION ACCESS FROM DATABASE TO USER

$sudo mysgl -u root -p

1. GRANT SPECIFIED DATABASE AND TABLE TO USER
   mysql>grant select on databaseName.tablesName to 'nameUser'@'localhost';
   mysql>grant insert on databaseName.tablesName to 'nameUser'@'localhost';
   mysql>grant update on databaseName.tablesName to 'nameUser'@'localhost';
   mysql>grant delete on databaseName.tablesName to 'nameUser'@'localhost';
   mysql>grant select, insert, update, delete on databaseName.tablesName to 'nameUser'@'localhost';
   
2. GRANT SPECIFIED DATABASE AND ALL TABLES TO USER
   mysql>grant select on databaseName.* to 'nameUser'@'localhost';
   
3. SHOW ALL PERMISSION ACCESS
   mysql>show grants;

4. SHOW SPECIFIED PERMISSION ACCESS
   mysql>show grant for 'nameUser'@'localhost';
   
5. ALLOW REMOTE ACCESS SPECIFIED DATABASE AND TABLE TO MySQL User
   mysql>grant all privileges on databaseName.tablesName to 'nameUser'@'localhost';

6. ALLOW REMOTE ACCESS SPECIFIED DATABASE AND ALL TABLES TO MySQL User
   mysql>grant all privileges on databaseName.* to 'nameUser'@'localhost';

7. ALLOW REMOTE ACCESS ALL DATABASE AND ALL TABLES TO MySQL User
   mysql>grant all privileges on *.* to 'nameUser'@'localhost';
 
8. DELETE SPECIFIED DATABASE AND TABLES FROM USER
   mysql>revoke all privileges on databaseName.tablesName from 'nameUser'@'localhost';
   mysql>drop user 'nameUser'@'localhost';
   
9. DELETE SPECIFIED DATABASE AND ALL TABLES FROM USER
   mysql>revoke all privileges on databaseName.* from 'nameUser'@'localhost';
   mysql>drop user 'nameUser'@'localhost';
   
10. DELETE ALL DATABASE AND ALL TABLES FROM USER
   mysql>revoke all privileges on *.* from 'nameUser'@'localhost';
   mysql>drop user 'nameUser'@'localhost';
